58 void precomputeColors() {
59 static const std::vector<std::tuple<double, double, double>> baseColors = {
60 {0.267004, 0.004874, 0.329415},
61 {0.229739, 0.322361, 0.545706},
62 {0.127568, 0.566949, 0.550556},
63 {0.369214, 0.788888, 0.382914},
64 {0.993248, 0.906157, 0.143936}
65 };
66 int numSteps = 256; // Increase for finer gradients
67 precomputedColors.resize(numSteps);
68
69 for (int i = 0; i < numSteps; ++i) {
70 float value = static_cast<float>(i) / static_cast<float>(numSteps - 1);
71 int index = static_cast<int>(value * (static_cast<float>(baseColors.size()) - 1));
72 float fraction = (value * (static_cast<float>(baseColors.size()) - 1)) - static_cast<float>(index);
73 auto [r1, g1, b1] = baseColors[index];
74 auto [r2, g2, b2] = baseColors[std::min(index + 1, static_cast<int>(baseColors.size() - 1))];
75
76 float red = r1 + fraction * (r2 - r1);
77 float green = g1 + fraction * (g2 - g1);
78 float blue = b1 + fraction * (b2 - b1);
79 precomputedColors[i] = {red, green, blue, 1.0};
80 }
81 }
